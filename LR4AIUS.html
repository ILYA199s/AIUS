def _countplot(df, pad):
    """
          

    (pandas.core.frame.DataFrame, int) -> None

    :
        df - 
        pad -      

    : None
    """

    #   
    plt.figure(
        figsize = (10, 7) #    
    )

    ax = plt.axes() #     
    ax.xaxis.tick_bottom() #     

    #     
    ax = sns.countplot(df.label, label = '')
    
    counts = df.groupby(df.label)['label'].count().index.tolist()

    i = 0

    for p in ax.patches:
        height = p.get_height()

        ax.text(
            p.get_x() + p.get_width() / 2.0, # X   
            height + pad, # Y   
            df.label.value_counts()[counts[i]], # 
            ha = 'center', # 
            fontdict = {
                'fontsize': 14, #  
                'color': '#000000' #  
            },
        )

        i += 1

    #    
    ax.tick_params(
        axis = 'x', # 
        direction = 'out', #   
        length = 10, #   
        width = 1, #    
        color = '#000000', #   
        pad = 5, #      
        labelsize = 14, #  
        labelcolor = '#000000', #  
        bottom = True,  #   
    )
    ax.tick_params(
        axis = 'y', # 
        direction = 'out', #   
        length = 10, #   
        width = 1, #    
        color = '#000000', #   
        pad = 5, #      
        labelsize = 14, #  
        labelcolor = '#000000', #  
        left = True  #   
    )

    #  
    ax.set_title(
        label = '      ', # 
        fontdict = {
            'fontsize': 18, #  
            'color': '#000000' #  
        },
        pad = 10 #     
    )

    #     
    ax.set_xlabel(
        '',
        fontsize = 14, #  
        fontdict = {
            'color': '#000000' #  
        },
        labelpad = 10 # 

    );
    ax.set_ylabel(
        '',
        fontsize = 14, #  
        fontdict = {
            'color': '#000000' #  
        },
        labelpad = 10 # 
    );
    
    plt.show() #
